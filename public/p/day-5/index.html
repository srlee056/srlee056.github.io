def solution(scoville, K):
    answer = 0
    # heapify, heappop, heappush ÌôúÏö©
    heapq.heapify(scoville) #min heap Íµ¨ÏÑ±
    while True:
        s1 = heapq.heappop(scoville)
        if s1 >= K : #Î™®Îì† Ïä§ÏΩîÎπå ÏßÄÏàòÍ∞Ä KÏù¥ÏÉÅ
            break
        elif len(scoville) == 0: # KÏù¥ÏÉÅÏúºÎ°ú ÎßåÎì§ Ïàò ÏóÜÎäî Í≤ΩÏö∞
            answer = -1

        answer += 1
    return answer

def solution(tickets):
    routes = {}
    for a, b in tickets: #Ï∂úÎ∞úÏßÄ, ÎèÑÏ∞©ÏßÄ
        routes[a] = routes.get(a, [])
