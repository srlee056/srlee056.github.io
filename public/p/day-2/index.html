      def concat(self, L):
    lastNode = self.tail.prev
    firstNode = L.head.next
    lastNode.next = firstNode
    firstNode.prev = lastNode
    self.tail = L.tail
    self.nodeCount += L.nodeCount

        def postfixEval(tokenList):
    valStack = ArrayStack()

    for t in tokenList:
        if type(t) is int:
            valStack.push(t)
        else:
            b = valStack.pop()
            a = valStack.pop()
            if t == '*':
                valStack.push(a*b)
            elif t == '/':
                valStack.push(a/b)
            elif t == '+':
                valStack.push(a+b)
            elif t == '-':
                valStack.push(a-b)

    return valStack.pop()
