    def solution(number, k):
    collected = []
    for i, num in enumerate(number):
        # collectedÏóê ÎØ∏Î¶¨ Îì§Ïñ¥Í∞Ñ ÏõêÏÜåÍ∞Ä ÏûàÏùÑ Í≤É
